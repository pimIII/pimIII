CRIA O REPO NO GITHUB
https://github.com/andersonraujjo/html_lab.git
------------------------------------------------------------------
DA O CD PASTA NO CMD
------------------------------------------------------------------
DA O GIT INIT
------------------------------------------------------------------
DA O GIT REMOTE ADD ORIGIN
git remote add origin https://github.com/andersonraujjo/html_lab.git
------------------------------------------------------------------
DA O GIT ADD
git add .
------------------------------------------------------------------
COMITTA
git commit -m "escreve aqui"
------------------------------------------------------------------
DA O PUSH COM O -U ORIGIN MAIN (SÓ DA PRIMEIRA VEZ)
git push -u origin main
------------------------------------------------------------------
SE PRECISAR RENOMEIA/DEFINE DE MASTER PRA MAIN
git branch -M main
------------------------------------------------------------------
FORÇA O PUSH
git push -u origin main --force
------------------------------------------------------------------
SETA O NOVO NOME DO REPO
git remote set-url origin https://github.com/seu-usuario/novo-nome-do-repo.git
------------------------------------------------------------------
PRA CONFERIR
git remote -v
------------------------------------------------------------------
MOSTRAR BRANCH ATUAL
git branch --show-current
------------------------------------------------------------------
MUDA PRA SANDBOX
git checkout sandbox
